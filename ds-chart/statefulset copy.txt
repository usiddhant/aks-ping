apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: ds
spec:
  serviceName: ds
  replicas: {{ .Values.replicaCount }}
  selector:
    matchLabels:
      app: ds
  template:
    metadata:
      labels:
        app: ds
    spec:
      initContainers:
        - name: ds-init
          image: "{{ .Values.image.repository }}:{{ .Values.image.tag }}"
          imagePullPolicy: {{ .Values.image.pullPolicy }}

          command:
            - /bin/sh
            - -c
            - |
              echo "Checking if DS is already initialized..."
              ls -lrt /opt/opendj
              ls -lrt /opt/opendj/data
              ls -lrt /opt/opendj/db

              if [ -d "/opt/opendj/data/db" ] && [ "$(ls -A /opt/opendj/data/db 2>/dev/null)" ]; then
                echo "DS already initialized, skipping setup-profile"
                exit 0
              fi

              echo "Reading passwords from secrets..."
              ADMIN_PWD=$(cat /opt/opendj/secrets/admin.pwd)
              cd /opt/opendj/bin

              echo "Running AM CTS profile..."
              ./setup-profile \
                --profile am-cts \
                --set am-cts/tokenExpirationPolicy:am-sessions-only \
                --set am-cts/amCtsAdminPassword:${ADMIN_PWD}

              echo "Running AM Config profile..."
              ./setup-profile \
                --profile am-config \
                --set am-config/amConfigAdminPassword:${ADMIN_PWD}

              echo "Running AM Identity Store profile..."
              ./setup-profile \
                --profile am-identity-store \
                --set am-identity-store/amIdentityStoreAdminPassword:${ADMIN_PWD}

              echo "All profiles applied successfully"
              cat /opt/opendj/config/profiles.version
              echo "End"
              ls -lrt /opt/opendj
              echo "End2"
              ls -lrt /opt/opendj/db
              cp -ar /opt/opendj/db /opt/opendj/data/
              cp -ar /opt/opendj/config/schema /opt/opendj/data/db/
              echo "End3"
              ls -lrt /opt/opendj/data
              echo "End4"

          volumeMounts:
            - name: data
              mountPath: /opt/opendj/data
            - name: secrets
              mountPath: /opt/opendj/secrets
              readOnly: true
  
      containers:
        - name: ds
          image: "{{ .Values.image.repository }}:{{ .Values.image.tag }}"
          imagePullPolicy: {{ .Values.image.pullPolicy }}

          args:
            - start-ds

          env:
            - name: DS_SET_UID_ADMIN_AND_MONITOR_PASSWORDS
              value: "true"
            - name: DS_UID_ADMIN_PASSWORD_FILE
              value: /opt/opendj/secrets/admin.pwd
            - name: DS_UID_MONITOR_PASSWORD_FILE
              value: /opt/opendj/secrets/monitor.pwd

          ports:
            - containerPort: 1389
            - containerPort: 1636
            - containerPort: 4444
            - containerPort: 8989

          volumeMounts:
            - name: data
              mountPath: /opt/opendj/data
            - name: secrets
              mountPath: /opt/opendj/secrets
              readonly: true

          resources:
            {{- toYaml .Values.resources | nindent 12 }}

      volumes:
        - name: secrets
          secret:
            secretName: ds-secrets

  volumeClaimTemplates:
    - metadata:
        name: data
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: {{ .Values.persistence.size }}





            initContainers:
        - name: ds-init
          image: "{{ .Values.image.repository }}:{{ .Values.image.tag }}"
          imagePullPolicy: {{ .Values.image.pullPolicy }}
          command:
          - sh
          - -c
          - "/opt/opendj/runtime-scripts/init-and-restore.sh || /opt/opendj/default-scripts/init-and-restore.sh"
          env:
            - name: DS_SET_UID_ADMIN_AND_MONITOR_PASSWORDS
              value: "true"
            - name: DS_UID_ADMIN_PASSWORD_FILE
              value: /opt/opendj/secrets/admin.pwd
          volumeMounts:
            - name: data
              mountPath: /opt/opendj/data
            - name: secrets
              mountPath: /opt/opendj/secrets
